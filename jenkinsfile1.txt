pipeline {
   agent any
   	stages {
		stage('Initiation') {
			steps {
				echo "This initiation stage of pipeline"
				sh "git init"
			}
		}
        	stage('CreateNewFile') {
			steps {
		                sh "cd ~/git_repo"
		                sh "touch WelcomeJenkinsFile2.txt"
		                sh "ls > WelcomeJenkinsFile2.txt"
			}
		}
		stage('AddFileToGit') {
			steps {
				sh "git add ."
			}
		}
		stage('CommitandPushFileToGit') {
			steps {
			     sh '''#!/bin/bash
				git commit -m "New Commit 3"
				git push -u origin master
				echo "Stop"
				''' 
			}
		}
		stage('lastStage') {
			steps {
				echo "Last stage of Pipeline"
				
			}
		}
	}
}
