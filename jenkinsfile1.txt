pipeline {
   agent any
   	stages {
		stage('Initiation') {
			steps {
				echo "This initiation stage of pipeline"
			}
		}
        	stage('CreateNewFile') {
			steps {
		                sh "cd ~/git_repo"
		                sh "touch WelcomeJenkinsFile2.txt"
		                sh "ls > WelcomeJenkinsFile2.txt"
			}
		}
		stage('AddFileToGit') {
			steps {
				sh "git add WelcomeJenkinsFile2.txt"
			}
		}
		stage('CommitFileToGit') {
			steps {
			     sh '''#!/bin/bash
				 git commit -m "New Commit 2"
				 '''
			    
			}
		}
		stage('PushFileToGit') {
			steps {
				sh '''#!/bin/bash
				git push -u origin master
				echo "Stop"
				'''
			}
		}
	}
}
