pipeline {
   agent any
   	stages {
		stage('Initiation') {
			steps {
				echo "This initiation stage of pipeline"
			}
		}
        	stage('CreateNewFile') {
			steps {
		                sh "cd ~/git_repo"
		                sh "touch WelcomeJenkinsFile2.txt"
		                sh "ls > WelcomeJenkinsFile2.txt"
			}
		}
		stage('AddFileToGit') {
			steps {
				sh "git add ."
			}
		}
		stage('CommitFileToGit') {
			steps {
			     sh '''#!/bin/bash
				git commit -m "New Commit 3"
				''' 
			}
		}
		stage('Push File To Github') {
			steps {
				sh "git push origin HEAD:master"
				
			}
		}
		stage('lastStage') {
			steps {
				echo "Last stage of Pipeline"
				
			}
		}
	}
}
